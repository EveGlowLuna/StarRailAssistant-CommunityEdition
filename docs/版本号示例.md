# 版本号使用示例

## 版本号判断逻辑

### 判断是否为测试版
```javascript
function isBetaVersion(version) {
  const parts = version.split('.');
  const lastPart = parseInt(parts[parts.length - 1]);
  return lastPart !== 0;
}

// 示例
isBetaVersion('0.1.0'); // false - 正式版
isBetaVersion('0.1.1'); // true - 测试版
isBetaVersion('0.2.0'); // false - 正式版
isBetaVersion('0.2.1'); // true - 测试版
```

### 版本号比较
```javascript
function compareVersions(v1, v2) {
  const parts1 = v1.split('.').map(Number);
  const parts2 = v2.split('.').map(Number);
  
  const maxLen = Math.max(parts1.length, parts2.length);
  
  for (let i = 0; i < maxLen; i++) {
    const p1 = parts1[i] || 0;
    const p2 = parts2[i] || 0;
    
    if (p1 > p2) return 1;
    if (p1 < p2) return -1;
  }
  
  return 0;
}

// 示例
compareVersions('0.1.1', '0.1.0'); // 1 (0.1.1 > 0.1.0)
compareVersions('0.2.0', '0.1.1'); // 1 (0.2.0 > 0.1.1)
compareVersions('0.1.0', '0.1.0'); // 0 (相等)
```
